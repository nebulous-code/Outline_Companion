<%*
// ============================================================================
// Outline Companion — Debug
// ============================================================================
// Validates your Outline Companion configuration and writes a timestamped
// report to your configured debug output folder. The report file is
// temporary and can be deleted at any time.
//
// Run this from the Obsidian command palette:
//   "Templater: Insert Outline Companion Debug"
// ============================================================================

const {
    CHAPTER_FOLDER,
    METADATA_FOLDER,
    METADATA_PREFIX,
    SPLIT_CHAR,
    NOTICE_DURATION,
    LOG_LEVEL,
    CREATE_DIRECTION,
    PANE_FALLBACK,
    CHAPTER_TEMPLATE,
    METADATA_TEMPLATE,
    DEBUG_OUTPUT_FOLDER,
} = tp.user.OutlineCompanionConfig();

// ─── SETUP ───────────────────────────────────────────────────────────────────

const VALID_LOG_LEVELS = ["silent", "normal", "verbose", "debug"];
const VALID_CREATE_DIRECTIONS = ["both", "metadata-only", "chapter-only"];
const VALID_PANE_FALLBACKS = ["new-tab", "none"];

const timestamp = new Date();
const pad = n => String(n).padStart(2, "0");
const ts = `${timestamp.getFullYear()}${pad(timestamp.getMonth()+1)}${pad(timestamp.getDate())}${pad(timestamp.getHours())}${pad(timestamp.getMinutes())}${pad(timestamp.getSeconds())}`;
const reportFilename = `${ts}_Outline_Companion_Debug_Results`;
const reportFolder = DEBUG_OUTPUT_FOLDER && DEBUG_OUTPUT_FOLDER !== "" ? DEBUG_OUTPUT_FOLDER : "/";
const reportPath = reportFolder === "/" 
    ? `${reportFilename}.md`
    : `${reportFolder}/${reportFilename}.md`;

const lines = [];
const issues = [];

function pass(label, detail = "") {
    lines.push(`- ✅ **${label}**${detail ? " — " + detail : ""}`);
}

function fail(label, detail = "") {
    lines.push(`- ❌ **${label}**${detail ? " — " + detail : ""}`);
    issues.push(`${label}${detail ? ": " + detail : ""}`);
}

function info(label, detail = "") {
    lines.push(`- ℹ️ **${label}**${detail ? " — " + detail : ""}`);
}

function section(title) {
    lines.push(`\n## ${title}\n`);
}

// ─── REPORT HEADER ───────────────────────────────────────────────────────────

lines.push(`> [!note] Temporary File`);
lines.push(`> This file was generated by Outline Companion Debug and can be deleted at any time. It will not affect any scripts or settings.`);
lines.push(``);
lines.push(`# Outline Companion Debug Report`);
lines.push(``);
lines.push(`**Run at:** ${timestamp.toLocaleString()}`);
lines.push(``);

// ─── CONFIG VALUES ───────────────────────────────────────────────────────────

section("Config Values");
info("CHAPTER_FOLDER", `\`${CHAPTER_FOLDER}\``);
info("METADATA_FOLDER", `\`${METADATA_FOLDER}\``);
info("METADATA_PREFIX", `\`${METADATA_PREFIX}\``);
info("SPLIT_CHAR", SPLIT_CHAR === "" ? "`(empty — full filename used as key)`" : `\`${SPLIT_CHAR}\``);
info("NOTICE_DURATION", NOTICE_DURATION === null ? "`null` (Obsidian default ~5s)" : NOTICE_DURATION === 0 ? "`0` (stays open until clicked)" : `\`${NOTICE_DURATION}\` ms (${NOTICE_DURATION/1000}s)`);
info("LOG_LEVEL", `\`${LOG_LEVEL}\``);
info("CREATE_DIRECTION", `\`${CREATE_DIRECTION}\``);
info("PANE_FALLBACK", `\`${PANE_FALLBACK}\``);
info("CHAPTER_TEMPLATE", CHAPTER_TEMPLATE === "" ? "`(empty — blank files created)`" : `\`${CHAPTER_TEMPLATE}\``);
info("METADATA_TEMPLATE", METADATA_TEMPLATE === "" ? "`(empty — blank files created)`" : `\`${METADATA_TEMPLATE}\``);
info("DEBUG_OUTPUT_FOLDER", DEBUG_OUTPUT_FOLDER === "" ? "`(empty — vault root)`" : `\`${DEBUG_OUTPUT_FOLDER}\``);

// ─── CONFIG VALIDATION ───────────────────────────────────────────────────────

section("Config Validation");

CHAPTER_FOLDER && CHAPTER_FOLDER !== ""
    ? pass("CHAPTER_FOLDER is set")
    : fail("CHAPTER_FOLDER is empty", "Set this to your manuscript folder path");

METADATA_FOLDER && METADATA_FOLDER !== ""
    ? pass("METADATA_FOLDER is set")
    : fail("METADATA_FOLDER is empty", "Set this to your metadata folder path");

METADATA_PREFIX && METADATA_PREFIX !== ""
    ? pass("METADATA_PREFIX is set")
    : fail("METADATA_PREFIX is empty", "Set this to the prefix used on your metadata files");

VALID_LOG_LEVELS.includes(LOG_LEVEL)
    ? pass("LOG_LEVEL is valid", `"${LOG_LEVEL}"`)
    : fail("LOG_LEVEL is invalid", `"${LOG_LEVEL}" is not one of: ${VALID_LOG_LEVELS.join(", ")}`);

VALID_CREATE_DIRECTIONS.includes(CREATE_DIRECTION)
    ? pass("CREATE_DIRECTION is valid", `"${CREATE_DIRECTION}"`)
    : fail("CREATE_DIRECTION is invalid", `"${CREATE_DIRECTION}" is not one of: ${VALID_CREATE_DIRECTIONS.join(", ")}`);

VALID_PANE_FALLBACKS.includes(PANE_FALLBACK)
    ? pass("PANE_FALLBACK is valid", `"${PANE_FALLBACK}"`)
    : fail("PANE_FALLBACK is invalid", `"${PANE_FALLBACK}" is not one of: ${VALID_PANE_FALLBACKS.join(", ")}`);

if (NOTICE_DURATION !== null && typeof NOTICE_DURATION !== "number") {
    fail("NOTICE_DURATION is invalid", "Must be a number in milliseconds or null");
} else {
    pass("NOTICE_DURATION is valid");
}

// ─── FOLDER CHECKS ───────────────────────────────────────────────────────────

section("Folder Checks");

const allFiles = app.vault.getMarkdownFiles();

const chapterFolderExists = app.vault.getAbstractFileByPath(CHAPTER_FOLDER) !== null;
chapterFolderExists
    ? pass("Chapter folder exists", `\`${CHAPTER_FOLDER}/\``)
    : fail("Chapter folder not found", `\`${CHAPTER_FOLDER}/\` does not exist in the vault`);

if (chapterFolderExists) {
    const chapterFiles = allFiles.filter(f => f.parent?.path === CHAPTER_FOLDER);
    info("Chapter files found", `${chapterFiles.length} markdown file(s)`);
    chapterFiles.slice(0, 5).forEach(f => lines.push(`  - \`${f.basename}\``));
    if (chapterFiles.length > 5) lines.push(`  - *(and ${chapterFiles.length - 5} more)*`);
}

const metadataFolderExists = app.vault.getAbstractFileByPath(METADATA_FOLDER) !== null;
metadataFolderExists
    ? pass("Metadata folder exists", `\`${METADATA_FOLDER}/\``)
    : fail("Metadata folder not found", `\`${METADATA_FOLDER}/\` does not exist in the vault`);

if (metadataFolderExists) {
    const metadataFiles = allFiles.filter(f => f.parent?.path === METADATA_FOLDER);
    info("Metadata files found", `${metadataFiles.length} markdown file(s)`);
    metadataFiles.slice(0, 5).forEach(f => lines.push(`  - \`${f.basename}\``));
    if (metadataFiles.length > 5) lines.push(`  - *(and ${metadataFiles.length - 5} more)*`);
}

// ─── TEMPLATE FILE CHECKS ────────────────────────────────────────────────────

section("Template File Checks");

if (CHAPTER_TEMPLATE === "") {
    info("Chapter template", "Not configured — new chapters will be created as blank files");
} else {
    const chapterTemplateFile = app.vault.getAbstractFileByPath(CHAPTER_TEMPLATE);
    chapterTemplateFile
        ? pass("Chapter template found", `\`${CHAPTER_TEMPLATE}\``)
        : fail("Chapter template not found", `\`${CHAPTER_TEMPLATE}\` does not exist — new chapters will fall back to blank files`);
}

if (METADATA_TEMPLATE === "") {
    info("Metadata template", "Not configured — new metadata files will be created as blank files");
} else {
    const metadataTemplateFile = app.vault.getAbstractFileByPath(METADATA_TEMPLATE);
    metadataTemplateFile
        ? pass("Metadata template found", `\`${METADATA_TEMPLATE}\``)
        : fail("Metadata template not found", `\`${METADATA_TEMPLATE}\` does not exist — new metadata files will fall back to blank files`);
}

// ─── WORKSPACE LAYOUT ────────────────────────────────────────────────────────

section("Workspace Layout");

function describeLeaf(leaf, indent) {
    const prefix = "  ".repeat(indent);
    if (typeof leaf.openFile === "function") {
        const title = leaf.view?.file?.basename ?? "(no file open)";
        lines.push(`${prefix}- 📄 Tab: \`${title}\``);
    } else {
        lines.push(`${prefix}- *(unknown leaf type)*`);
    }
}

function describeNode(node, indent = 0) {
    const prefix = "  ".repeat(indent);
    if (!node) {
        lines.push(`${prefix}- *(null node)*`);
        return;
    }
    if (typeof node.openFile === "function") {
        describeLeaf(node, indent);
        return;
    }
    const type = node.type ?? node.constructor?.name ?? "unknown";
    const childCount = node.children?.length ?? 0;
    lines.push(`${prefix}- **${type}** (${childCount} child${childCount !== 1 ? "ren" : ""})`);
    if (node.children) {
        node.children.forEach(child => describeNode(child, indent + 1));
    }
}

const root = app.workspace.rootSplit;
lines.push(`Root split type: \`${root.type}\`, children: \`${root.children?.length ?? 0}\``);
lines.push(``);
describeNode(root);

// ─── PANE TARGETING TEST ─────────────────────────────────────────────────────

section("Pane Targeting Test");

function getLeafFromTabGroup(tabGroup) {
    if (!tabGroup?.children?.length) return null;
    const active = tabGroup.children.find(leaf => leaf === tabGroup.activeLeaf)
        ?? tabGroup.children[0];
    return (typeof active?.openFile === "function") ? active : null;
}

const leftTabGroup = root?.children?.[0];
const leftLeaf = getLeafFromTabGroup(leftTabGroup);
leftLeaf
    ? pass("Left pane (chapter target) — leaf found", `currently showing: \`${leftLeaf.view?.file?.basename ?? "no file"}\``)
    : fail("Left pane (chapter target) — leaf not found", "The script will fall back to PANE_FALLBACK behavior");

const rightSplit = root?.children?.[1];
const topRightTabGroup = rightSplit?.children?.[0];
const topRightLeaf = getLeafFromTabGroup(topRightTabGroup);
topRightLeaf
    ? pass("Top-right pane (metadata target) — leaf found", `currently showing: \`${topRightLeaf.view?.file?.basename ?? "no file"}\``)
    : fail("Top-right pane (metadata target) — leaf not found", "The script will fall back to PANE_FALLBACK behavior. Check that your layout has a right column split into top and bottom panes.");

// ─── SUMMARY ─────────────────────────────────────────────────────────────────

section("Summary");

if (issues.length === 0) {
    lines.push(`✅ **All checks passed.** Outline Companion is configured correctly.`);
} else {
    lines.push(`❌ **${issues.length} issue(s) found:**`);
    lines.push(``);
    issues.forEach((issue, i) => lines.push(`${i + 1}. ${issue}`));
    lines.push(``);
    lines.push(`Address the issues above and re-run this script to verify.`);
}

// ─── WRITE REPORT ────────────────────────────────────────────────────────────

const reportContent = lines.join("\n");

try {
    await app.vault.create(reportPath, reportContent);
    const reportFile = tp.file.find_tfile(reportFilename);
    if (reportFile) {
        await app.workspace.getLeaf("tab").openFile(reportFile);
    }
    new Notice(`Outline Companion Debug: Report written to "${reportPath}"`, NOTICE_DURATION ?? undefined);
} catch (e) {
    new Notice(`Outline Companion Debug: Failed to write report to "${reportPath}". Check that the folder exists.`, NOTICE_DURATION ?? undefined);
}
%>
